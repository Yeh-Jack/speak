cd /workspaces/education/backend

# 1. Run tests.
uv run pytest tests/unit/ -v

# 2. Start the backend server.
uv run uvicorn app.main:app --reload --host 0.0.0.0 --port 8000

# 3. Get video information
curl -X POST http://localhost:8000/api/v1/videos/info \
  -H "Content-Type: application/json" \
  -d '{"youtube_url": "https://www.youtube.com/watch?v=SPGn9MZb1a0"}'

# 4. Create video from YouTube (test with a short video)
curl -X POST http://localhost:8000/api/v1/videos/youtube \
  -H "Content-Type: application/json" \
  -d '{"youtube_url": "https://www.youtube.com/watch?v=SPGn9MZb1a0", "chunk_duration": 30}'

# 5. List videos
curl http://localhost:8000/api/v1/videos

# 6. Delete video
curl -X DELETE http://localhost:8000/api/v1/videos/{video_id}} \
  -H "Content-Type: application/json"

This will:
1. Download the video
2. Create sentence-snapped chunks
3. Generate transcript
4. Return the complete video object

# 7. Check video status
curl http://localhost:8000/api/v1/videos/{video_id}

# 8. Set video downloading status:
curl -X PATCH "http://localhost:8000/api/v1/videos/{video_id}" \
  -H "Content-Type: application/json" \
  -d '{"status": "chunking_complete"}'

# 9. Retry downloading:
curl -X POST "http://localhost:8000/api/v1/videos/{video_id}/retry"



# 6. Get video chunks
curl http://localhost:8000/api/v1/videos/{video_id}/chunks
